<script>
var div = $('#mydiv')[0];

function myFunction()
{
vowels = ['a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U']
var content = document.getElementById("target").value;
var output = "";
for (var i = 0; i < content.length; i++ )
{
var letter = content.charAt(i);
if(letter == "\n"){
output += "<br>";
} else if(letter == " "){
output += letter;
} else if($.inArray(letter, vowels) > -1){
var redFont = "<span class=\"red\">"+letter+"</span>";
output += redFont;
} else {
var blueFont = "<span class=\"blue\">"+letter+"</span>";
output += blueFont;
}
}
document.getElementById("mydiv").innerHTML = output;
}

$('#size').change(function(){
var sizeSelected = $(this).val();
$('#mydiv p').css('font-size',sizeSelected );
$('#mydiv').css('font-size',sizeSelected );

});

function clearTextBox(){
document.getElementById("mydiv").innerHTML = "";
document.getElementById("target").value = "";

}
